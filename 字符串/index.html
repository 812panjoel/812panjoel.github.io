[start:] 从start索引开始到结尾"pyehzmzbrhtwgzm"[8:]"rhtwgzm"
[:end] 从头开始到end-1"pyehzmzbrhtwgzm"[:8]"pyehzmzb"
[start:end] 从start到end"pyehzmzbrhtwgzm"[3:6]"hzm"
[start:end:step] 从start提取到end，步长是step"pyehzmzbrhtwgzm"[1:6:2]"yhm"
